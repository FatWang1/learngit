install:
	sudo apt-get install git
	./config
	make
	sudo make install
make a git repository:
	mkdir dir_name
	cd dir_name
init git repository:
	git init
add file:
	touch file_name(and input content)
	git add file_name(can add lots of files,use space as sign)
	git commit -m "what did you do"
look for the status:
	git status
look for the changes:
	git diff
look for change logs:
	git log(can +"--pretty=oneline"in the end to simply the output)
back to the lastest version:
	git reset --hard HEAD^
back to the which version:
	git reset --hard (commit id at last 5 of head number)
look for commit id history:
	git reflog
back to the lastest git commit(or add):
	git checkout -- file_name
remove file:
	git rm file_name
	git commit -m "remove file_name"
remove dir:
	git rm -r dir_name
	git commit -m "remove dir_name"
renew the file(had been removed):
	git checkout -- file_name
link git-net-server:
	1.set ssh
	2.press:git push -u origin ssh_link
push local to net:
	git push -u origin master
modify and push:
	git push orgin master
clone:
	git clone ssh_link
creat a new branch and move to this branch:
	git checkout -b branch_name
	=
	git branch branch_name
	+
	git checkout dev
list all branches:
	git branch
switch branch:
	git checkout branch_name
mix other branch to this branch:
	git merge branch_name
remove branch:
	git branch -d branch_name
mix other branch to this branch(without fast forward):
	git merge --no-ff -m "merge with no-ff" branch_name
look for branch`s history:
	git log --graph --pretty=oneline --abbrev-commit
store workplace:
	git stash
look for workplace:
	git stash list
go on the workplace:
	git stash pop
	OR
	git stash apply(back to previous workplace)
	+
	git stash drop(remove stash)
remove a bransh which never be mixed:
	git branch -D branch_name
